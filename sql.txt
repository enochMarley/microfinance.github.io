CREATE DATABASE microfinance;

CREATE TABLE microfinance.admin ( adminId int(11) NOT NULL AUTO_INCREMENT, adminUsername text NOT NULL, adminPassword text NOT NULL, PRIMARY KEY (adminId) );

CREATE TABLE microfinance.clients ( Id int(11) NOT NULL AUTO_INCREMENT, fullName text NOT NULL, birthDate text NOT NULL, nationality text NOT NULL, emailAddress text NOT NULL, gender text NOT NULL, phoneNumber text NOT NULL, resAddress text NOT NULL, nextOfKin text NOT NULL, accountNumber text NOT NULL, signUpdate timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, balance double NOT NULL DEFAULT 0, debit double NOT NULL DEFAULT 0, profileImg text NOT NULL, PRIMARY KEY (Id) );

CREATE TABLE microfinance.loans ( Id int(11) NOT NULL AUTO_INCREMENT, fullName text NOT NULL, birthDate text NOT NULL, nationality text NOT NULL, emailAddress text NOT NULL, gender text NOT NULL, phoneNumber text NOT NULL, resAddress text NOT NULL, guarantor text NOT NULL, amount double NOT NULL, interest double NOT NULL, signUpdate timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, profileImg text NOT NULL, debit double NOT NULL DEFAULT 0, PRIMARY KEY (Id) );


CREATE TABLE microfinance.transactions ( id int(11) NOT NULL AUTO_INCREMENT, clientId int(11) NOT NULL, transaction text NOT NULL, transactionDate timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (id) );

CREATE TABLE employees (Id int(11) NOT NULL AUTO_INCREMENT,fullName text NOT NULL,birthDate text NOT NULL,nationality text NOT NULL,emailAddress text NOT NULL,gender text NOT NULL,phoneNumber text NOT NULL,resAddress text NOT NULL,salary double NOT NULL,signUpdate timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,profileImg text NOT NULL,PRIMARY KEY (Id)
);
